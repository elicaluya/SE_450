001    /**
002     * Utility class for Inventory.  Fields are mutable and package-private.
003     * Does not override <code>equals</code> or <code>hashCode</code>.
004     *
005     * @objecttype Mutable Data Class
006     */
007    final class Record {
008      /**
009       * The video.
010       * @invariant <code>video != null</code>
011       */
012      VideoObj video;
013      /**
014       * The number of copies of the video that are in the inventory.
015       * @invariant <code>numOwned > 0</code>
016       */
017      int numOwned;
018      /**
019       * The number of copies of the video that are currently checked out.
020       * @invariant <code>numOut <= numOwned</code>
021       */
022      int numOut;
023      /**
024       * The total number of times this video has ever been checked out.
025       * @invariant <code>numRentals >= numOut</code>
026       */
027      int numRentals;
028    
029      /**
030       * Initialize all object attributes.
031       */
032      Record(VideoObj video, int numOwned, int numOut, int numRentals) {
033        this.video = video;
034        this.numOwned = numOwned;
035        this.numOut = numOut;
036        this.numRentals = numRentals;
037      }
038      /**
039       * Return a shallow copy of this record.
040       */
041      public Record copy() {
042        return new Record(video,numOwned,numOut,numRentals);
043      }
044      /**
045       * Return a string representation of the object in the following format:
046       * <code>"video [numOwned,numOut,numRentals]"</code>.
047       */
048      public String toString() {
049        StringBuffer buffer = new StringBuffer();
050        buffer.append(video);
051        buffer.append(" [");
052        buffer.append(numOwned);
053        buffer.append(",");
054        buffer.append(numOut);
055        buffer.append(",");
056        buffer.append(numRentals);
057        buffer.append("]");
058        return buffer.toString();
059      }
060    }

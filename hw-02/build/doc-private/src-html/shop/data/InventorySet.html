105      /**
106       * Implementation of Record interface.
107       *
108       * <p>This is a utility class for Inventory.  Fields are mutable and
109       * package-private.</p>
110       *
111       * <p><b>Class Invariant:</b> No two instances may reference the same Video.</p>
112       *
113       * @see Record
114       */
115      private static final class RecordObj implements Record {
116        Video video; // the video
117        int numOwned;   // copies owned
118        int numOut;     // copies currently rented
119        int numRentals; // total times video has been rented
120        
121        RecordObj(Video video, int numOwned, int numOut, int numRentals) {
122          this.video = video;
123          this.numOwned = numOwned;
124          this.numOut = numOut;
125          this.numRentals = numRentals;
126        }
127        public Video video() {
128          return video;
129        }
130        public int numOwned() {
131          return numOwned;
132        }
133        public int numOut() {
134          return numOut;
135        }
136        public int numRentals() {
137          return numRentals;
138        }
139        public String toString() {
140          StringBuffer buffer = new StringBuffer();
141          buffer.append(video);
142          buffer.append(" [");
143          buffer.append(numOwned);
144          buffer.append(",");
145          buffer.append(numOut);
146          buffer.append(",");
147          buffer.append(numRentals);
148          buffer.append("]");
149          return buffer.toString();
150        }

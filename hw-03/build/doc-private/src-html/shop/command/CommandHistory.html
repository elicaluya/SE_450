001    package shop.command;
002    
003    /**
004     * An unbounded list of commands with undo/redo capability.
005     *
006     * <p>Logically, the functionality is described in terms of two stacks:
007     * <code>undoable</code> and <code>redoable</code>, both initially empty.</p>
008     */
009    public interface CommandHistory {
010      /**
011       * Add command <code>undoable</code> and clear <code>redoable</code>.
012       * @param cmd the command to be run.
013       */
014      public void add(UndoableCommand cmd);
015      
016      /**
017       * Returns a <code>RerunnableCommand</code> that, when run does the following:
018       * Pop command from <code>undoable</code>, undo it, then push it
019       * onto <code>redoable</code>.
020       * @throws EmptyStackException if there is no undoable command.
021       */
022      public RerunnableCommand getUndo();
023    
024      /**
025       * Returns a <code>RerunnableCommand</code> that, when run does the following:
026       * Pop command from <code>redoable</code>, redo it, then push it
027       * onto <code>undoable</code>.
028       * @throws EmptyStackException if there is no redoable command.
029       */
030      public RerunnableCommand getRedo();
031    }

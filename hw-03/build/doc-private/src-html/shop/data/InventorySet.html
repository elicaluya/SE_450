066      /**
067       * Add or remove copies of a video from the inventory.
068       * If a video record is not already present (and change is
069       * positive), a record is created. 
070       * If a record is already present, <code>numOwned</code> is
071       * modified using <code>change</code>.
072       * If <code>change</code> brings the number of copies to be less
073       * than one, the record is removed from the inventory.
074       * @param video the video to be added.
075       * @param change the number of copies to add (or remove if negative).
076       * @return A copy of the previous record for this video (if any)
077       * @throws IllegalArgumentException if video null or change is zero
078       */
079      Record addNumOwned(Video video, int change) {
080        if (video == null || change == 0)
081          throw new IllegalArgumentException();
082        
083        RecordObj r = (RecordObj) _data.get(video);
084        if (r == null && change < 1) {
085          throw new IllegalArgumentException();
086        } else if (r == null) {
087          _data.put(video, new RecordObj(video, change, 0, 0));
088        } else if (r.numOwned+change < r.numOut) {
089          throw new IllegalArgumentException();
090        } else if (r.numOwned+change < 1) {
091          _data.remove(video);
092        } else {
093          _data.put(video, new RecordObj(video, r.numOwned + change, r.numOut, r.numRentals));
094        }
095        return r;
096      }

152      /**
153       * Implementation of Record interface.
154       *
155       * <p>This is a utility class for Inventory.  Fields are mutable and
156       * package-private.</p>
157       *
158       * <p><b>Class Invariant:</b> No two instances may reference the same Video.</p>
159       *
160       * @see Record
161       */
162      private static final class RecordObj implements Record {
163        Video video; // the video
164        int numOwned;   // copies owned
165        int numOut;     // copies currently rented
166        int numRentals; // total times video has been rented
167        
168        RecordObj(Video video, int numOwned, int numOut, int numRentals) {
169          this.video = video;
170          this.numOwned = numOwned;
171          this.numOut = numOut;
172          this.numRentals = numRentals;
173        }
174        RecordObj copy() {
175          return new RecordObj(video, numOwned, numOut, numRentals);
176        }
177        public Video video() {
178          return video;
179        }
180        public int numOwned() {
181          return numOwned;
182        }
183        public int numOut() {
184          return numOut;
185        }
186        public int numRentals() {
187          return numRentals;
188        }
189        public boolean equals(Object thatObject) {
190          return video.equals(((Record)thatObject).video());
191        }
192        public int hashCode() {
193          return video.hashCode();
194        }
195        public String toString() {
196          StringBuffer buffer = new StringBuffer();
197          buffer.append(video);

201          buffer.append(numOut);
202          buffer.append(",");

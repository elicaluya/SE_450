001    package shop.data;
002    
003    import shop.command.UndoableCommand;
004    import java.util.Map;
005    
006    /**
007     * Implementation of command to clear the inventory.
008     * @see Data
009     */
010    final class CmdClear implements UndoableCommand {
011      private InventorySet _inventory;
012      private Map _oldvalue;
013      CmdClear(InventorySet inventory) {
014        _inventory = inventory;
015      }
016      public boolean run() {
017        if (_oldvalue != null)
018          return false;
019        try {
020          _oldvalue = _inventory.clear();
021          _inventory.getHistory().add(this);
022          return true;
023        } catch (ClassCastException e) {
024          return false;
025        }
026      }
027      public void undo() {
028        _inventory.replaceMap(_oldvalue);
029      }
030      public void redo() {
031        _inventory.clear();
032      }
033    }

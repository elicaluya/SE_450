031       * <ul>
032       * <li>If a video record is not already present (and change is
033       * positive), a record is created.</li>
034       * <li>If a record is already present, <code>numOwned</code> is

036       * <li>If <code>change</code> brings the number of copies to be less

038       * </ul>
039       * @param video the video to be added.
040       * @param change the number of copies to add (or remove if negative).

042       */
043      static public UndoableCommand newAddCmd(Inventory inventory, Video video, int change) {
044        if (!(inventory instanceof InventorySet))
045          throw new IllegalArgumentException();
046        return new CmdAdd((InventorySet) inventory, video, change);
047      }

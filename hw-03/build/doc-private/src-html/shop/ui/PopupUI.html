006    final class PopupUI implements UI {
007      PopupUI() {}
008    
009      public void displayMessage(String message) {
010        JOptionPane.showMessageDialog(null,message);
011      }
012    
013      public void displayError(String message) {
014        JOptionPane.showMessageDialog(null,message,"Error",JOptionPane.ERROR_MESSAGE);
015      }
016    
017      public void processMenu(UIMenu menu) {
018        StringBuffer b = new StringBuffer();
019        b.append(menu.getHeading());
020        b.append("\n");
021        b.append("Enter choice by number:");
022        b.append("\n");
023    
024        for (int i = 1; i < menu.size(); i++) {
025          b.append("  " + i + ". " + menu.getPrompt(i));
026          b.append("\n");
027        }
028    
029        String response = JOptionPane.showInputDialog(b.toString());
030        int selection;
031        try {
032          selection = Integer.parseInt(response, 10);
033          if ((selection < 0) || (selection >= menu.size()))
034            selection = 0;
035        } catch (NumberFormatException e) {
036          selection = 0;
037        }
038    
039        menu.runAction(selection);
040      }
